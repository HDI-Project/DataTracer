def get_solvers():
    """Get a list of the available datatracer solvers.

    Returns:
        list:
            List of the names of the available datatracer solvers.
    """
    return [
        pretrained_solver[:-3]
        for pretrained_solver in os.listdir(PRETRAINED_DIR)
        if pretrained_solver.endswith('.dt')
    ]
